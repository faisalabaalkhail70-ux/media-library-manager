class MainWindow(QMainWindow):
    def __init__(self) -> None:
        super().__init__()
        self.setWindowTitle("Media Library Manager")
        self.resize(1400, 850)

        central = QWidget()
        self.setCentralWidget(central)

        root = QHBoxLayout(central)
        root.setContentsMargins(0, 0, 0, 0)
        root.setSpacing(0)

        sidebar = QFrame()
        sidebar.setObjectName("sidebar")
        sidebar.setFixedWidth(230)

        side_layout = QVBoxLayout(sidebar)
        side_layout.setContentsMargins(12, 20, 12, 20)
        side_layout.setSpacing(8)

        logo = QLabel("Media Library\nManager")
        logo.setObjectName("h1")
        logo.setAlignment(Qt.AlignCenter)
        side_layout.addWidget(logo)

        self.stack = QStackedWidget()

        self.views = [
            ("Dashboard", DashboardView()),
            ("Scanner", ScannerView()),
            ("Library", LibraryView()),
            ("Movies", MoviesView()),
            ("Shows", ShowsView()),
            ("Duplicates", DuplicatesView()),
            ("Rename", RenameView()),
            ("Reports", ReportsView()),
            ("Settings", SettingsView()),
        ]

        for _, view in self.views:
            self.stack.addWidget(view)

        self.nav_buttons = []
        for index, (label, _) in enumerate(self.views):
            btn = QPushButton(label)
            btn.setObjectName("sidebar_btn")
            btn.setCheckable(True)
            btn.clicked.connect(lambda checked, i=index: self.switch_view(i))
            side_layout.addWidget(btn)
            self.nav_buttons.append(btn)

        side_layout.addStretch()

        root.addWidget(sidebar)
        root.addWidget(self.stack, 1)

        self.switch_view(0)

    def switch_view(self, index: int) -> None:
        self.stack.setCurrentIndex(index)

        for i, btn in enumerate(self.nav_buttons):
            btn.setChecked(i == index)

        current = self.stack.currentWidget()

        if hasattr(current, "load_rows"):
            current.load_rows()
        elif hasattr(current, "refresh"):
            current.refresh()
        elif hasattr(current, "load_settings"):
            current.load_settings()

class DuplicatesView(QWidget):
    def __init__(self) -> None:
        super().__init__()
        self.repo = DuplicatesRepository()
        self.model = DuplicatesModel([])
        self.worker = None

        layout = QVBoxLayout(self)

        title = QLabel("Duplicates")
        title.setObjectName("h1")
        layout.addWidget(title)

        subtitle = QLabel("Find exact and possible duplicate media files.")
        subtitle.setObjectName("muted")
        layout.addWidget(subtitle)

        toolbar = QHBoxLayout()

        self.refresh_btn = QPushButton("Refresh")
        self.refresh_btn.clicked.connect(self.load_rows)
        toolbar.addWidget(self.refresh_btn)

        self.find_btn = QPushButton("Find Duplicates")
        self.find_btn.setObjectName("primary")
        self.find_btn.clicked.connect(self.find_duplicates)
        toolbar.addWidget(self.find_btn)

        toolbar.addStretch()
        layout.addLayout(toolbar)

        self.status_label = QLabel("Ready.")
        self.status_label.setObjectName("muted")
        layout.addWidget(self.status_label)

        self.table = QTableView()
        self.table.setModel(self.model)
        self.table.setAlternatingRowColors(True)
        self.table.verticalHeader().setVisible(False)
        self.table.horizontalHeader().setStretchLastSection(True)
        layout.addWidget(self.table)

        self.load_rows()

    def load_rows(self) -> None:
        rows = self.repo.fetch_duplicate_rows()
        self.model.set_rows(rows)
        self.status_label.setText(f"Loaded {len(rows)} duplicate rows.")

    def find_duplicates(self) -> None:
        if self.worker and self.worker.isRunning():
            QMessageBox.information(self, "Scan in progress", "A duplicate scan is already running.")
            return

        self.refresh_btn.setEnabled(False)
        self.find_btn.setEnabled(False)
        self.status_label.setText("Scanning for duplicates...")

        self.worker = DuplicateWorker()
        self.worker.finished_build.connect(self.on_finished)
        self.worker.failed.connect(self.on_failed)
        self.worker.start()

    def on_finished(self, result: dict) -> None:
        self.refresh_btn.setEnabled(True)
        self.find_btn.setEnabled(True)
        self.load_rows()

        self.status_label.setText(
            f'Duplicate scan finished. Exact groups: {result["exact_groups"]}, '
            f'Possible groups: {result["possible_groups"]}'
        )

        QMessageBox.information(
            self,
            "Duplicate scan complete",
            f'Exact groups: {result["exact_groups"]}\nPossible groups: {result["possible_groups"]}'
        )

    def on_failed(self, message: str) -> None:
        self.refresh_btn.setEnabled(True)
        self.find_btn.setEnabled(True)
        self.status_label.setText("Duplicate scan failed.")
        QMessageBox.critical(self, "Duplicate scan failed", message)

class RenameView(QWidget):
    def __init__(self) -> None:
        super().__init__()
        self.rename_service = RenameService()
        self.model = RenamePreviewModel([])
        self.worker = None
        self.undo_worker = None

        layout = QVBoxLayout(self)

        title = QLabel("Rename")
        title.setObjectName("h1")
        layout.addWidget(title)

        subtitle = QLabel("Preview and apply renames using your configured template.")
        subtitle.setObjectName("muted")
        layout.addWidget(subtitle)

        toolbar = QHBoxLayout()

        self.template_input = QLineEdit()
        self.template_input.setPlaceholderText("{Title} ({Year}){Ext}")
        toolbar.addWidget(self.template_input)

        self.preview_btn = QPushButton("Preview")
        self.preview_btn.clicked.connect(self.preview)
        toolbar.addWidget(self.preview_btn)

        self.apply_btn = QPushButton("Apply")
        self.apply_btn.setObjectName("primary")
        self.apply_btn.clicked.connect(self.apply_changes)
        toolbar.addWidget(self.apply_btn)

        self.undo_btn = QPushButton("Undo Latest")
        self.undo_btn.clicked.connect(self.undo_latest)
        toolbar.addWidget(self.undo_btn)

        layout.addLayout(toolbar)

        self.table = QTableView()
        self.table.setModel(self.model)
        self.table.setAlternatingRowColors(True)
        self.table.verticalHeader().setVisible(False)
        self.table.horizontalHeader().setStretchLastSection(True)
        layout.addWidget(self.table)

        self.preview()

    def preview(self) -> None:
        rows = self.rename_service.build_preview(self.template_input.text().strip())
        self.model.set_rows(rows)

    def apply_changes(self) -> None:
        rows = [r for r in self.model.rows() if r["status"] == "valid"]
        if not rows:
            QMessageBox.information(self, "Nothing to apply", "No valid rename operations.")
            return

        self.preview_btn.setEnabled(False)
        self.apply_btn.setEnabled(False)
        self.undo_btn.setEnabled(False)

        self.worker = RenameWorker(rows)
        self.worker.finished_apply.connect(self.on_applied)
        self.worker.failed.connect(self.on_failed)
        self.worker.start()

    def on_applied(self, result: dict) -> None:
        self.preview_btn.setEnabled(True)
        self.apply_btn.setEnabled(True)
        self.undo_btn.setEnabled(True)

        QMessageBox.information(
            self,
            "Rename complete",
            f'Success: {result["success"]}, Failed: {result["failed"]}'
        )
        self.preview()

    def undo_latest(self) -> None:
        self.preview_btn.setEnabled(False)
        self.apply_btn.setEnabled(False)
        self.undo_btn.setEnabled(False)

        self.undo_worker = UndoWorker(limit=20)
        self.undo_worker.finished_undo.connect(self.on_undo_complete)
        self.undo_worker.failed.connect(self.on_failed)
        self.undo_worker.start()

    def on_undo_complete(self, result: dict) -> None:
        self.preview_btn.setEnabled(True)
        self.apply_btn.setEnabled(True)
        self.undo_btn.setEnabled(True)

        QMessageBox.information(
            self,
            "Undo complete",
            f'Undone: {result["undone"]}, Failed: {result["failed"]}'
        )
        self.preview()

    def on_failed(self, message: str) -> None:
        self.preview_btn.setEnabled(True)
        self.apply_btn.setEnabled(True)
        self.undo_btn.setEnabled(True)
        QMessageBox.critical(self, "Rename failed", message)

    def _fetch_show_rows(self) -> list[dict]:
        with get_connection() as conn:
            rows = conn.execute(
                """
                SELECT
                    me.id AS entity_id,
                    me.title AS show_title,
                    me.release_year,
                    me.tmdb_id,
                    COUNT(CASE WHEN ep.is_missing = 0 AND ep.media_file_id IS NOT NULL THEN 1 END) AS present_episodes,
                    COUNT(CASE WHEN ep.is_missing = 1 THEN 1 END) AS missing_episodes,
                    COUNT(ep.id) AS total_episode_rows
                FROM media_entities me
                LEFT JOIN episodes ep ON ep.entity_id = me.id
                WHERE me.media_type = 'show'
                GROUP BY me.id, me.title, me.release_year, me.tmdb_id
                ORDER BY me.title COLLATE NOCASE
                """
            ).fetchall()

        return [dict(r) for r in rows]

    def _fetch_all_missing_rows(self) -> list[dict]:
        return self.repo.fetch_all_missing_rows()

    def _filter_missing_rows_for_show(self, entity_id: int) -> list[dict]:
        return [row for row in self._all_missing_rows if row.get("entity_id") == entity_id]

    def load_rows(self) -> None:
        self._show_rows = self._fetch_show_rows()
        self._all_missing_rows = self._fetch_all_missing_rows()
        self._selected_show_id = None

        self.populate_shows_table(self._show_rows)
        self.model.set_rows(self._all_missing_rows)

        self.status_label.setText(
            f"Loaded {len(self._show_rows)} shows and {len(self._all_missing_rows)} missing episode rows."
        )

        if self._show_rows:
            self.shows_table.selectRow(0)
            self.on_show_selection_changed()
        else:
            self.show_summary_label.setText("No tracked series found.")
            self.missing_label.setText("No missing episode rows loaded.")

    def populate_shows_table(self, rows: list[dict]) -> None:
        self.shows_table.setRowCount(len(rows))

        for row_index, row in enumerate(rows):
            values = [
                row.get("show_title") or "Unknown",
                str(row.get("release_year") or ""),
                str(row.get("tmdb_id") or ""),
                str(row.get("present_episodes") or 0),
                str(row.get("missing_episodes") or 0),
                str(row.get("total_episode_rows") or 0),
            ]

            for col_index, value in enumerate(values):
                item = QTableWidgetItem(value)
                if col_index != 0:
                    item.setTextAlignment(Qt.AlignCenter)
                item.setData(Qt.UserRole, row)
                self.shows_table.setItem(row_index, col_index, item)

        self.shows_table.resizeColumnsToContents()

    def on_show_selection_changed(self) -> None:
        row_index = self.shows_table.currentRow()
        if row_index < 0 or row_index >= len(self._show_rows):
            self._selected_show_id = None
            self.show_summary_label.setText("No show selected.")
            self.missing_label.setText("All missing episodes across tracked series.")
            self.model.set_rows(self._all_missing_rows)
            return

        row = self._show_rows[row_index]
        self._selected_show_id = row.get("entity_id")

        show_title = row.get("show_title") or "Unknown"
        year = row.get("release_year") or "—"
        tmdb_id = row.get("tmdb_id") or "—"
        present = row.get("present_episodes") or 0
        missing = row.get("missing_episodes") or 0
        total = row.get("total_episode_rows") or 0

        self.show_summary_label.setText(
            f"Selected series: {show_title} | Year: {year} | TMDB ID: {tmdb_id} | "
            f"Present: {present} | Missing: {missing} | Total tracked rows: {total}"
        )

        filtered_rows = self._filter_missing_rows_for_show(self._selected_show_id)
        self.model.set_rows(filtered_rows)
        self.missing_label.setText(
            f"Missing episodes currently listed for: {show_title} ({len(filtered_rows)} rows)"
        )

    def show_all_missing_rows(self) -> None:
        self._selected_show_id = None
        self.shows_table.clearSelection()
        self.model.set_rows(self._all_missing_rows)
        self.show_summary_label.setText("No show selected.")
        self.missing_label.setText(
            f"All missing episodes across tracked series ({len(self._all_missing_rows)} rows)"
        )

            )

            if len(show_names) > 15:
                summary += "\n..."

            QMessageBox.information(self, "Check complete", summary)
        except Exception as exc:
            self.status_label.setText("Missing episode check failed.")
